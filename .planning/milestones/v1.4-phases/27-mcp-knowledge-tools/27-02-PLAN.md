         const slug = resolveProjectFromCwd(cwd, config);

         // Get dependsOn from config
         const allProjects = [...(config.projects ?? []), ...(config.multiCopy ?? []).flatMap(m => /* flatten */)];
         const projectEntry = allProjects.find(p => p.slug === slug);
         const relatedProjects = projectEntry?.dependsOn ?? [];

         // Get convention violations from health findings
         const findings = getActiveFindings(getInstance().db, slug);
         const violations = findings.filter(f => f.checkType === "convention_violation").length;

         // Get staleness
         const knowledge = getKnowledge(getInstance().db, slug);
         const stalenessScore = knowledge
           ? computeStalenessScore(knowledge.lastModified, knowledge.commitsSinceUpdate)
           : null;
         const staleKnowledge = stalenessScore !== null && stalenessScore < 50;

         return c.json({ slug, relatedProjects, violations, staleKnowledge, stalenessScore });
       })

       ```bash
       #!/bin/bash
       # Knowledge digest -- SessionStart hook
       # Calls MC API /api/knowledge/digest and formats knowledge banner
       # Shows AFTER risks digest. Zero output when no knowledge context.

       MC_API="${MC_API_URL:-http://100.123.8.125:3000}"
       CWD="${PWD}"

       # URL-encode the cwd for query parameter
       ENCODED_CWD=$(python3 -c "import urllib.parse; print(urllib.parse.quote('$CWD'))")

       DATA=$(curl -sf --max-time 5 "$MC_API/api/knowledge/digest?cwd=$ENCODED_CWD" 2>/dev/null) || exit 0

       python3 - "$DATA" << 'PYEOF'
       import json
       import sys

       try:
           data = json.loads(sys.argv[1])
       except (json.JSONDecodeError, IndexError):
           sys.exit(0)

       slug = data.get("slug")
       if not slug:
           sys.exit(0)

       related = data.get("relatedProjects", [])
       violations = data.get("violations", 0)
       stale = data.get("staleKnowledge", False)

       # Zero noise when nothing to report (per D-04: concise, 3-5 lines max)
       if not related and violations == 0 and not stale:
           sys.exit(0)

       parts = []
       if related:
           parts.append(f"Related: {', '.join(related)}")
       if violations > 0:
           parts.append(f"Convention violations: {violations}")
       if stale:
           score = data.get("stalenessScore", "?")
           parts.append(f"CLAUDE.md stale (freshness: {score}/100)")

       print(f"KNOWLEDGE ({slug}):")
       for part in parts:
           print(f"  {part}")
       PYEOF
       ```
