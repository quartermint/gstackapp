**1. Create CaptureIntent.swift** (per D-03, IOS-04):
```swift
import AppIntents

struct CaptureIntent: AppIntent {
    static var title: LocalizedStringResource = "Quick Capture"
    static var openAppWhenRun: Bool = true

    func perform() async throws -> some IntentResult {
        UserDefaults(suiteName: AppConstants.appGroupID)?
            .set(true, forKey: "showQuickCapture")
        return .result()
    }
}
```
This completes within the 3-second WidgetKit execution budget because it only writes a boolean to UserDefaults (per IOS-04).

```swift
func testQuickCaptureCreatesQueuedCapture() throws {
    let container = try ModelContainer(
        for: QueuedCapture.self,
        configurations: ModelConfiguration(isStoredInMemoryOnly: true)
    )
    let context = ModelContext(container)
    let capture = QueuedCapture(rawContent: "Quick thought", type: "text", sourceApp: "widget")
    context.insert(capture)
    try context.save()

    let descriptor = FetchDescriptor<QueuedCapture>()
    let captures = try context.fetch(descriptor)
    XCTAssertEqual(captures.count, 1)
    XCTAssertEqual(captures.first?.rawContent, "Quick thought")
    XCTAssertEqual(captures.first?.type, "text")
    XCTAssertEqual(captures.first?.sourceApp, "widget")
    XCTAssertEqual(captures.first?.syncStatus, "pending")
}
```
