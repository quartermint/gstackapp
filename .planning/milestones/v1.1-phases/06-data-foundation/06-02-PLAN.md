export const projectHealth = sqliteTable("project_health", {
  id: integer("id").primaryKey({ autoIncrement: true }),
  projectSlug: text("project_slug").notNull(),

  severity: text("severity").notNull(),
  detail: text("detail").notNull(),

  index("health_slug_check_idx").on(table.projectSlug, table.checkType),

**GREEN phase — implement `packages/api/src/db/queries/health.ts`:**

Functions to implement:

1. `upsertHealthFinding(db: DrizzleDb, sqlite: Database.Database, finding: HealthFindingInput): void`
   - Uses explicit transactional SELECT-then-UPDATE/INSERT (NOT `onConflictDoUpdate` — see research for why)
   - SELECT: `WHERE project_slug = ? AND check_type = ? AND resolved_at IS NULL`
   - If found: UPDATE severity, detail, metadata WHERE id = ? (NEVER update detected_at)
   - If not found: INSERT with `detected_at = new Date().toISOString()`
   - Wrap in `sqlite.transaction()`

2. `resolveFindings(sqlite: Database.Database, projectSlug: string, activeCheckTypes: string[]): void`
   - UPDATE `resolved_at = now()` WHERE `project_slug = ?` AND `resolved_at IS NULL` AND `check_type NOT IN (activeCheckTypes)`
   - If activeCheckTypes is empty, resolve ALL active findings for the project

3. `getActiveFindings(db: DrizzleDb, projectSlug?: string): HealthFinding[]`
   - SELECT from projectHealth WHERE resolvedAt IS NULL
   - Optionally filter by projectSlug
   - Return all matching rows

4. `getProjectRiskLevel(db: DrizzleDb, projectSlug: string): RiskLevel`
   - Get active findings for project
   - If none: return "healthy"
   - Map worst severity: "critical" -> "critical", "warning" -> "warning", "info" -> "healthy"
   - Note: "unmonitored" is NOT computed here — it's set at the API/scanner layer for GitHub-only projects. This function only computes from findings.

Import types: `HealthFindingInput` from `@mission-control/shared`, `Database` from `better-sqlite3`, `DrizzleDb` from `../index.js`.

Run tests — all must PASS.
