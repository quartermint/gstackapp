    ```swift
    import SwiftUI
    import SwiftData

    struct ShareView: View {
        let extensionContext: NSExtensionContext?
        @Environment(\.modelContext) private var modelContext
        @State private var extractedText: String = ""
        @State private var extractedURL: String?
        @State private var captureType: String = "text"
        @State private var isLoading = true

        var body: some View {
            NavigationStack {
                VStack(spacing: 16) {
                    if isLoading {
                        ProgressView("Reading content...")
                    } else {
                        // Content preview (per D-01)
                        VStack(alignment: .leading, spacing: 8) {
                            if let url = extractedURL {
                                Label(url, systemImage: "link")
                                    .font(.caption)
                                    .foregroundStyle(.secondary)
                                    .lineLimit(2)
                            }

                            Text(extractedText)
                                .font(.body)
                                .lineLimit(5)
                                .frame(maxWidth: .infinity, alignment: .leading)
                                .padding()
                                .background(Color(.systemGray6))
                                .clipShape(RoundedRectangle(cornerRadius: 8))
                        }

                        Spacer()
                    }
                }
                .padding()
                .navigationTitle("MC Capture")
                .navigationBarTitleDisplayMode(.inline)
                .toolbar {
                    ToolbarItem(placement: .cancellationAction) {
                        Button("Cancel") {
                            extensionContext?.completeRequest(returningItems: nil)
                        }
                    }
                    ToolbarItem(placement: .confirmationAction) {
                        Button("Save") {
                            saveCapture()
                        }
                        .bold()
                        .disabled(isLoading || extractedText.isEmpty)
                    }
                }
            }
            .task {
                await extractContent()
            }
        }

        private func extractContent() async {
            guard let items = extensionContext?.inputItems as? [NSExtensionItem] else {
                isLoading = false
                return
            }

            for item in items {
                guard let attachments = item.attachments else { continue }

                for attachment in attachments {
                    // Extract URL first (higher priority -- includes page title context)
                    if attachment.hasItemConformingToTypeIdentifier("public.url") {
                        if let url = try? await attachment.loadItem(forTypeIdentifier: "public.url") as? URL {
                            extractedURL = url.absoluteString
                            captureType = "link"
                            // Use attributed content text if available, otherwise use URL
                            if let attributedText = item.attributedContentText?.string, !attributedText.isEmpty {
                                extractedText = attributedText
                            } else {
                                extractedText = url.absoluteString
                            }
                            isLoading = false
                            return
                        }
                    }

                    // Extract plain text
                    if attachment.hasItemConformingToTypeIdentifier("public.plain-text") {
                        if let text = try? await attachment.loadItem(forTypeIdentifier: "public.plain-text") as? String {
                            extractedText = text
                            captureType = "text"
                            isLoading = false
                            return
                        }
                    }
                }
            }

            isLoading = false
        }

        private func saveCapture() {
            let capture = QueuedCapture(
                rawContent: extractedText,
                type: captureType,
                linkUrl: extractedURL
            )
            // IOS-11: sourceApp extracted from extensionContext host bundle ID if available
            // Note: not all host apps expose their bundle ID reliably
            modelContext.insert(capture)

            do {
                try modelContext.save()
            } catch {
                // Best-effort save -- if it fails, the capture is lost
                // In practice SwiftData auto-saves, explicit save is belt-and-suspenders
            }

            extensionContext?.completeRequest(returningItems: nil)
        }
    }
    ```

    **1. MissionControl/Services/SyncEngine.swift** (per D-11, D-12, IOS-07, IOS-08):

    ```swift
    import Foundation
    import SwiftData
    import UserNotifications

    @MainActor
    @Observable
    final class SyncEngine {
        private(set) var pendingCount: Int = 0
        private(set) var isSyncing: Bool = false
        private(set) var lastSyncAt: Date?

        private let apiClient: MCAPIClientProtocol
        private var modelContainer: ModelContainer?

        init(apiClient: MCAPIClientProtocol = MCAPIClient()) {
            self.apiClient = apiClient
        }

        func configure(container: ModelContainer) {
            self.modelContainer = container
            // Initial pending count
            updatePendingCount()
        }

        /// Flush all pending captures to the MC API.
        /// Guarded by isSyncing flag to prevent concurrent syncs (pitfall #5: scenePhase fires multiple times).
        /// Debounced by syncDebounceInterval to avoid hammering on rapid foreground transitions.
        func syncPendingCaptures() async {
            guard !isSyncing else { return }

            // Debounce: skip if last sync was less than 30s ago
            if let lastSync = lastSyncAt,
               Date().timeIntervalSince(lastSync) < AppConstants.syncDebounceInterval {
                return
            }

            guard let container = modelContainer else { return }

            isSyncing = true
            defer {
                isSyncing = false
                lastSyncAt = Date()
                updatePendingCount()
                updateBadgeCount()
            }

            let context = ModelContext(container)
            let descriptor = FetchDescriptor<QueuedCapture>(
                predicate: #Predicate<QueuedCapture> { capture in
                    capture.syncStatus == "pending" || capture.syncStatus == "failed"
                },
                sortBy: [SortDescriptor(\.capturedAt, order: .forward)]
            )

            guard let pending = try? context.fetch(descriptor) else { return }
            pendingCount = pending.count

            for capture in pending {
                // Skip items that have exceeded max retries
                guard capture.retryCount < AppConstants.maxRetryCount else {
                    continue
                }

                // Exponential backoff: skip if retried too recently
                if let lastRetry = capture.lastRetryAt, capture.retryCount > 0 {
                    let backoffInterval = pow(2.0, Double(capture.retryCount)) // 2s, 4s, 8s
                    if Date().timeIntervalSince(lastRetry) < backoffInterval {
                        continue
                    }
                }

                do {
                    // IOS-13: projectId is passed through to the API (user assignment preserved)
                    _ = try await apiClient.createCapture(
                        rawContent: capture.rawContent,
                        type: capture.type,
                        projectId: capture.projectId,
                        idempotencyKey: capture.idempotencyKey
                    )
                    capture.syncStatus = "synced"
                    pendingCount = max(0, pendingCount - 1)
                } catch {
                    capture.syncStatus = "failed"
                    capture.retryCount += 1
                    capture.lastRetryAt = Date()
                }
            }

            // Persist sync status changes
            try? context.save()
        }

        /// Update the pending count from the database (call after container is set or after sync).
        func updatePendingCount() {
            guard let container = modelContainer else { return }
            let context = ModelContext(container)
            let descriptor = FetchDescriptor<QueuedCapture>(
                predicate: #Predicate<QueuedCapture> { capture in
                    capture.syncStatus == "pending" || capture.syncStatus == "failed"
                }
            )
            pendingCount = (try? context.fetchCount(descriptor)) ?? 0
        }

        /// Update app icon badge to show pending capture count (IOS-10: D-10).
        private func updateBadgeCount() {
            let center = UNUserNotificationCenter.current()
            Task {
                let settings = await center.notificationSettings()
                if settings.badgeSetting == .enabled {
                    try? await center.setBadgeCount(pendingCount)
                }
            }
        }

        /// Request badge permission (call once on first app launch).
        func requestBadgePermission() {
            UNUserNotificationCenter.current().requestAuthorization(options: [.badge]) { _, _ in }
        }
    }
    ```
