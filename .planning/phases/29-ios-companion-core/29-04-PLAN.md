    **1. MissionControl/Services/LocationService.swift** (per IOS-11):

    City-level location service using kCLLocationAccuracyReduced. Requests permission on first launch. Share extension reads last known location from UserDefaults in the App Group (per research pitfall #6: don't request auth from extension).

    ```swift
    import Foundation
    import CoreLocation

    @MainActor
    @Observable
    final class LocationService: NSObject, CLLocationManagerDelegate {
        private(set) var currentCity: String?
        private let manager = CLLocationManager()
        private let geocoder = CLGeocoder()

        override init() {
            super.init()
            manager.delegate = self
            manager.desiredAccuracy = kCLLocationAccuracyReduced  // City-level only
        }

        /// Request when-in-use permission. Call on first app launch.
        func requestPermission() {
            manager.requestWhenInUseAuthorization()
        }

        /// Start monitoring significant location changes for city-level metadata.
        func startMonitoring() {
            manager.startMonitoringSignificantLocationChanges()
        }

        func stopMonitoring() {
            manager.stopMonitoringSignificantLocationChanges()
        }

        // MARK: - CLLocationManagerDelegate

        nonisolated func locationManager(_ manager: CLLocationManager, didUpdateLocations locations: [CLLocation]) {
            guard let location = locations.last else { return }
            Task { @MainActor in
                await self.reverseGeocode(location)
            }
        }

        nonisolated func locationManager(_ manager: CLLocationManager, didFailWithError error: Error) {
            // Location is supplementary -- failure is acceptable
        }

        // MARK: - Private

        private func reverseGeocode(_ location: CLLocation) async {
            do {
                let placemarks = try await geocoder.reverseGeocodeLocation(location)
                if let city = placemarks.first?.locality {
                    currentCity = city
                    // Persist to App Group UserDefaults so share extension can read it
                    let defaults = UserDefaults(suiteName: AppConstants.appGroupID)
                    defaults?.set(city, forKey: "lastKnownCity")
                }
            } catch {
                // Geocoding failure is non-fatal
            }
        }
    }
    ```

    **2. Update MissionControl/MissionControlApp.swift** (per D-11, D-19: foreground sync):

    Replace the shell with full wiring. Inject SyncEngine, ConnectionMonitor, LocationService into the environment. Trigger sync on scenePhase .active.

    ```swift
    import SwiftUI
    import SwiftData

    @main
    struct MissionControlApp: App {
        @Environment(\.scenePhase) private var scenePhase
        @State private var syncEngine = SyncEngine()
        @State private var connectionMonitor = ConnectionMonitor()
        @State private var locationService = LocationService()

        private let modelContainer: ModelContainer

        init() {
            do {
                modelContainer = try createSharedModelContainer()
            } catch {
                fatalError("Failed to create SwiftData container: \(error)")
            }
            syncEngine.configure(container: modelContainer)
        }

        var body: some Scene {
            WindowGroup {
                DashboardView(
                    syncEngine: syncEngine,
                    connectionMonitor: connectionMonitor
                )
                .onAppear {
                    connectionMonitor.start()
                    locationService.requestPermission()
                    locationService.startMonitoring()
                    syncEngine.requestBadgePermission()
                }
            }
            .modelContainer(modelContainer)
            .onChange(of: scenePhase) { _, newPhase in
                if newPhase == .active {
                    // IOS-07: Foreground sync (per D-11)
                    Task {
                        await connectionMonitor.probeHealth()
                        if connectionMonitor.isMCReachable {
                            await syncEngine.syncPendingCaptures()
                        } else {
                            syncEngine.updatePendingCount()
                        }
                    }
                }
            }
        }
    }
    ```
