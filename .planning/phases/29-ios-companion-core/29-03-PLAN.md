    ```swift
    import Foundation
    import Network

    @MainActor
    @Observable
    final class ConnectionMonitor {
        private(set) var isNetworkAvailable: Bool = true
        private(set) var isMCReachable: Bool = false
        private(set) var lastProbeAt: Date?

        private var monitor: NWPathMonitor?
        private let apiClient: MCAPIClientProtocol

        init(apiClient: MCAPIClientProtocol = MCAPIClient()) {
            self.apiClient = apiClient
        }

        func start() {
            let monitor = NWPathMonitor()
            self.monitor = monitor
            monitor.pathUpdateHandler = { [weak self] path in
                Task { @MainActor [weak self] in
                    guard let self else { return }
                    self.isNetworkAvailable = path.status == .satisfied
                    if path.status == .satisfied {
                        await self.probeHealth()
                    } else {
                        self.isMCReachable = false
                    }
                }
            }
            monitor.start(queue: DispatchQueue(label: "mc.network.monitor"))
        }

        func stop() {
            monitor?.cancel()
            monitor = nil
        }

        /// Probe the MC API health endpoint to confirm reachability over Tailscale.
        func probeHealth() async {
            do {
                let isOk = try await apiClient.probeHealth()
                isMCReachable = isOk
            } catch {
                isMCReachable = false
            }
            lastProbeAt = Date()
        }

        /// Open VPN settings so user can enable Tailscale (per research: no tailscale:// URL scheme).
        /// Falls back to instruction text if canOpenURL returns false.
        func openVPNSettings() {
            #if !targetEnvironment(simulator)
            if let url = URL(string: "App-Prefs:General&path=VPN") {
                if UIApplication.shared.canOpenURL(url) {
                    UIApplication.shared.open(url)
                    return
                }
            }
            #endif
            // Fallback: no-op on simulator or if App-Prefs is blocked.
            // UI should show text instructions as fallback.
        }
    }
    ```

    ```swift
    import Foundation
    import SwiftUI

    enum ProjectGroup: String, CaseIterable {
        case active = "Active"
        case idle = "Idle"
        case stale = "Stale"
    }

    @MainActor
    @Observable
    final class DashboardViewModel {
        private(set) var projects: [ProjectItem] = []
        private(set) var healthBySlug: [String: HealthCheckResponse] = [:]
        private(set) var isLoading: Bool = false
        private(set) var error: String?
        private(set) var lastRefreshAt: Date?

        private let apiClient: MCAPIClientProtocol

        init(apiClient: MCAPIClientProtocol = MCAPIClient()) {
            self.apiClient = apiClient
        }

        // MARK: - Grouped Projects (per D-05: Active/Idle/Stale)

        var activeProjects: [ProjectItem] {
            projects.filter { group(for: $0) == .active }
        }

        var idleProjects: [ProjectItem] {
            projects.filter { group(for: $0) == .idle }
        }

        var staleProjects: [ProjectItem] {
            projects.filter { group(for: $0) == .stale }
        }

        /// Group a project by its last commit date.
        /// Active: commit within 7 days. Idle: 7-30 days. Stale: >30 days or nil.
        func group(for project: ProjectItem) -> ProjectGroup {
            guard let dateString = project.lastCommitDate,
                  let date = ISO8601DateFormatter().date(from: dateString) else {
                return .stale
            }
            let daysSinceCommit = Calendar.current.dateComponents([.day], from: date, to: Date()).day ?? 999
            if daysSinceCommit <= 7 {
                return .active
            } else if daysSinceCommit <= 30 {
                return .idle
            } else {
                return .stale
            }
        }

        /// Worst severity for a project's health findings. Returns "healthy" if no findings.
        func worstSeverity(for slug: String) -> String {
            guard let response = healthBySlug[slug] else { return "healthy" }
            if response.findings.contains(where: { $0.severity == "critical" }) { return "critical" }
            if response.findings.contains(where: { $0.severity == "warning" }) { return "warning" }
            if response.findings.contains(where: { $0.severity == "info" }) { return "info" }
            return response.riskLevel ?? "healthy"
        }

        // MARK: - Data Fetching (per IOS-10: pull-to-refresh)

        func loadProjects() async {
            isLoading = true
            error = nil

            do {
                let fetched = try await apiClient.fetchProjects()
                projects = fetched
                lastRefreshAt = Date()

                // Fetch health for each project (batch)
                await withTaskGroup(of: (String, HealthCheckResponse?).self) { group in
                    for project in fetched {
                        group.addTask { [apiClient] in
                            let response = try? await apiClient.fetchHealthChecks(slug: project.slug)
                            return (project.slug, response)
                        }
                    }
                    for await (slug, response) in group {
                        if let response {
                            healthBySlug[slug] = response
                        }
                    }
                }
            } catch {
                self.error = error.localizedDescription
            }

            isLoading = false
        }

        // MARK: - Relative time formatting (per D-07: "Last synced: X ago")

        var lastRefreshRelativeText: String {
            guard let date = lastRefreshAt else { return "Never" }
            let formatter = RelativeDateTimeFormatter()
            formatter.unitsStyle = .abbreviated
            return formatter.localizedString(for: date, relativeTo: Date())
        }
    }
    ```
