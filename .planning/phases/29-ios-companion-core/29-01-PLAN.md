    **2. MissionControl/API/MCAPIClient.swift:**

    Create a protocol `MCAPIClientProtocol` and a concrete `MCAPIClient` class:

    ```swift
    import Foundation

    protocol MCAPIClientProtocol: Sendable {
        func createCapture(rawContent: String, type: String, projectId: String?, idempotencyKey: String) async throws -> CaptureItem
        func fetchProjects() async throws -> [ProjectItem]
        func fetchHealthChecks(slug: String) async throws -> HealthCheckResponse
        func probeHealth() async throws -> Bool
    }

    final class MCAPIClient: MCAPIClientProtocol {
        private let baseURL: URL
        private let session: URLSession
        private let decoder: JSONDecoder

        init(baseURL: URL = AppConstants.apiBaseURL, session: URLSession = .shared) {
            self.baseURL = baseURL
            self.session = session
            self.decoder = JSONDecoder()
        }

        func createCapture(rawContent: String, type: String, projectId: String?, idempotencyKey: String) async throws -> CaptureItem {
            let url = baseURL.appendingPathComponent("api/captures")
            var request = URLRequest(url: url)
            request.httpMethod = "POST"
            request.setValue("application/json", forHTTPHeaderField: "Content-Type")
            request.setValue(idempotencyKey, forHTTPHeaderField: "Idempotency-Key")

            let body = CreateCaptureRequest(
                rawContent: rawContent,
                type: type,
                projectId: projectId,
                clientId: "ios-app"
            )
            request.httpBody = try JSONEncoder().encode(body)

            let (data, response) = try await session.data(for: request)
            guard let httpResponse = response as? HTTPURLResponse,
                  httpResponse.statusCode == 201 else {
                throw MCAPIError.serverError(statusCode: (response as? HTTPURLResponse)?.statusCode ?? 0)
            }

            let captureResponse = try decoder.decode(CaptureResponse.self, from: data)
            return captureResponse.capture
        }

        func fetchProjects() async throws -> [ProjectItem] {
            let url = baseURL.appendingPathComponent("api/projects")
            let (data, response) = try await session.data(from: url)
            guard let httpResponse = response as? HTTPURLResponse,
                  httpResponse.statusCode == 200 else {
                throw MCAPIError.serverError(statusCode: (response as? HTTPURLResponse)?.statusCode ?? 0)
            }
            let projectResponse = try decoder.decode(ProjectListResponse.self, from: data)
            return projectResponse.projects
        }

        func fetchHealthChecks(slug: String) async throws -> HealthCheckResponse {
            let url = baseURL.appendingPathComponent("api/health-checks/\(slug)")
            let (data, response) = try await session.data(from: url)
            guard let httpResponse = response as? HTTPURLResponse,
                  httpResponse.statusCode == 200 else {
                throw MCAPIError.serverError(statusCode: (response as? HTTPURLResponse)?.statusCode ?? 0)
            }
            return try decoder.decode(HealthCheckResponse.self, from: data)
        }

        func probeHealth() async throws -> Bool {
            let url = baseURL.appendingPathComponent("api/health")
            var request = URLRequest(url: url, timeoutInterval: 5)
            request.httpMethod = "GET"
            let (data, response) = try await session.data(for: request)
            guard let httpResponse = response as? HTTPURLResponse,
                  httpResponse.statusCode == 200 else {
                return false
            }
            let probe = try decoder.decode(HealthProbeResponse.self, from: data)
            return probe.status == "ok"
        }
    }

    enum MCAPIError: Error, LocalizedError {
        case serverError(statusCode: Int)
        case decodingError(Error)

        var errorDescription: String? {
            switch self {
            case .serverError(let code): return "Server returned status \(code)"
            case .decodingError(let err): return "Decoding failed: \(err.localizedDescription)"
            }
        }
    }
    ```
