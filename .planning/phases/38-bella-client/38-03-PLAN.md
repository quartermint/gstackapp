       import { useChat } from "@ai-sdk/react";

       export function useBellaChat() {
         const API_BASE = import.meta.env.VITE_API_URL ?? "";

         return useChat({
           api: `${API_BASE}/api/chat`,
           id: "bella-chat",
           maxSteps: 5,
           onError: (error) => {

           },
         });
       }

       import { useBellaChat } from "../../hooks/use-bella-chat.js";
       import { BellaLayout } from "./bella-layout.js";
       import { ChatMessages } from "./chat-messages.js";
       import { ChatInput } from "./chat-input.js";
       import { ApiExplorer } from "./api-explorer.js";

       export function BellaChat() {
         const { messages, input, handleInputChange, handleSubmit, isLoading, error } = useBellaChat();
         const [explorerOpen, setExplorerOpen] = useState(false);

         return (
           <BellaLayout onExplorerToggle={() => setExplorerOpen(v => !v)}>
             <div className="flex flex-col h-full">
               <div className="flex-1 overflow-y-auto px-4 py-6">
                 <ChatMessages messages={messages} isLoading={isLoading} />
               </div>
               {error && (
                 <div className="px-4 py-2 text-sm text-rust">
                   {error.message.includes("503")
                     ? "MC's brain is warming up -- try again in a moment."
                     : `Error: ${error.message}`}
                 </div>
               )}
               <div className="border-t border-warm-gray/10 p-4">
                 <ChatInput
                   input={input}
                   onChange={handleInputChange}
                   onSubmit={handleSubmit}
                   isLoading={isLoading}
                 />
               </div>
             </div>
             <ApiExplorer open={explorerOpen} onClose={() => setExplorerOpen(false)} />
           </BellaLayout>
         );
       }
