```typescript
describe("cross-host dedup", () => {
  let instance: DatabaseInstance;

  beforeEach(() => {
    instance = createTestDb();
  });
  afterEach(() => {
    instance.sqlite.close();
  });

  it("getDiscoveriesByNormalizedUrl finds matching discoveries", () => {
    upsertDiscovery(instance.db, {
      path: "/Users/test/my-repo",
      host: "local",
      remoteUrl: "git@github.com:user/my-repo.git",
      name: "my-repo",
      lastCommitAt: null,
    });

    const { getDiscoveriesByNormalizedUrl } = require("../../db/queries/discoveries.js");
    // normalizeRemoteUrl("git@github.com:user/my-repo.git") = "github.com/user/my-repo"
    // normalizeRemoteUrl("https://github.com/user/my-repo") = "github.com/user/my-repo"
    const matches = getDiscoveriesByNormalizedUrl(instance.db, "github.com/user/my-repo");
    expect(matches).toHaveLength(1);
    expect(matches[0].host).toBe("local");
  });

  it("repos with no remote URL are never dedup matches", () => {
    upsertDiscovery(instance.db, {
      path: "/Users/test/no-remote",
      host: "local",
      remoteUrl: null,
      name: "no-remote",
      lastCommitAt: null,
    });

    const { getDiscoveriesByNormalizedUrl } = require("../../db/queries/discoveries.js");
    const matches = getDiscoveriesByNormalizedUrl(instance.db, "github.com/user/no-remote");
    expect(matches).toHaveLength(0);
  });

  it("same repo on local and github are deduped (second source skipped)", () => {
    // First: local discovery exists with SSH remote URL
    upsertDiscovery(instance.db, {
      path: "/Users/test/shared-repo",
      host: "local",
      remoteUrl: "git@github.com:user/shared-repo.git",
      name: "shared-repo",
      lastCommitAt: null,
    });

    // Now try GitHub discovery with HTTPS URL for same repo
    // The isAlreadyDiscoveredByRemoteUrl check should catch this
    // (tested via the query function since we can't easily invoke the full scan without mocking)
    const { getDiscoveriesByNormalizedUrl } = require("../../db/queries/discoveries.js");
    const { normalizeRemoteUrl } = require("../../services/git-health.js");

    const githubUrl = "https://github.com/user/shared-repo";
    const normalized = normalizeRemoteUrl(githubUrl);
    const matches = getDiscoveriesByNormalizedUrl(instance.db, normalized);

    // Should find the local discovery as a match
    expect(matches).toHaveLength(1);
    expect(matches[0].host).toBe("local");

    // Verify URL normalization works for dedup
    expect(normalizeRemoteUrl("git@github.com:user/shared-repo.git")).toBe(normalized);
  });
});
```
