    import type Database from "better-sqlite3";

    export interface VectorSearchResult {
      contentHash: string;
      distance: number;
    }

    /**
     * Two-step KNN query per D-10: vector query first (no JOINs),
     * enrich with metadata second.
     * Step 1: Returns content_hash + distance from vec0 table.
     */
    export function vectorKnnSearch(
      sqlite: Database.Database,
      queryEmbedding: Float32Array,
      limit: number = 20
    ): VectorSearchResult[] {
      const stmt = sqlite.prepare(`
        SELECT content_hash, distance
        FROM vec_search_index
        WHERE embedding MATCH ?
          AND k = ?
      `);
      const rows = stmt.all(queryEmbedding.buffer, limit) as Array<{
        content_hash: string;
        distance: number;
      }>;
      return rows.map(r => ({
        contentHash: r.content_hash,
        distance: r.distance,
      }));
    }

    /**
     * Check if content hash already has an embedding.
     */
    export function hasEmbedding(
      sqlite: Database.Database,
      contentHash: string
    ): boolean {
      const row = sqlite.prepare(
        'SELECT 1 FROM search_content WHERE content_hash = ? AND embedded = 1'
      ).get(contentHash);
      return !!row;
    }

    /**
     * Insert embedding into vec0 table and mark search_content as embedded.
     * Uses a transaction for atomicity.
     */
    export function insertEmbedding(
      sqlite: Database.Database,
      contentHash: string,
      embedding: Float32Array
    ): void {
      const insertVec = sqlite.prepare(
        'INSERT INTO vec_search_index(content_hash, embedding) VALUES (?, ?)'
      );
      const markEmbedded = sqlite.prepare(
        'UPDATE search_content SET embedded = 1 WHERE content_hash = ?'
      );
      const txn = sqlite.transaction(() => {
        insertVec.run(contentHash, embedding.buffer);
        markEmbedded.run(contentHash);
      });
      txn();
    }

    /**
     * Upsert content into search_content registry.
     * Returns true if this is new content (needs embedding), false if duplicate.
     */
    export function upsertSearchContent(
      sqlite: Database.Database,
      data: {
        contentHash: string;
        content: string;
        sourceType: string;
        sourceId: string;
        projectSlug: string | null;
        createdAt: string;
      }
    ): boolean {
      const existing = sqlite.prepare(
        'SELECT 1 FROM search_content WHERE content_hash = ?'
      ).get(data.contentHash);
      if (existing) return false;

      sqlite.prepare(
        `INSERT INTO search_content(content_hash, content, source_type, source_id, project_slug, created_at, embedded)
         VALUES (?, ?, ?, ?, ?, ?, 0)`
      ).run(data.contentHash, data.content, data.sourceType, data.sourceId, data.projectSlug, data.createdAt);
      return true;
    }

    /**
     * Step 2 of two-step pattern: enrich vector results with metadata.
     */
    export function enrichVectorResults(
      sqlite: Database.Database,
      contentHashes: string[]
    ): Map<string, { content: string; sourceType: string; sourceId: string; projectSlug: string | null; createdAt: string }> {
      const map = new Map();
      if (contentHashes.length === 0) return map;

      const placeholders = contentHashes.map(() => '?').join(',');
      const rows = sqlite.prepare(
        `SELECT content_hash, content, source_type, source_id, project_slug, created_at
         FROM search_content WHERE content_hash IN (${placeholders})`
      ).all(...contentHashes) as Array<{
        content_hash: string;
        content: string;
        source_type: string;
        source_id: string;
        project_slug: string | null;
        created_at: string;
      }>;

      for (const row of rows) {
        map.set(row.content_hash, {
          content: row.content,
          sourceType: row.source_type,
          sourceId: row.source_id,
          projectSlug: row.project_slug,
          createdAt: row.created_at,
        });
      }
      return map;
    }
    ```

    import type Database from "better-sqlite3";

    export interface EmbeddingJob {
      id: number;
      contentHash: string;
      status: string;
      attempts: number;
      maxAttempts: number;
      error: string | null;
      createdAt: string;
      updatedAt: string;
    }

    export function enqueueEmbeddingJob(sqlite: Database.Database, contentHash: string): void {
      const now = new Date().toISOString();
      sqlite.prepare(
        `INSERT INTO embedding_jobs(content_hash, status, attempts, max_attempts, created_at, updated_at)
         VALUES (?, 'pending', 0, 3, ?, ?)`
      ).run(contentHash, now, now);
    }

    export function claimPendingJobs(sqlite: Database.Database, batchSize: number = 10): EmbeddingJob[] {
      const now = new Date().toISOString();
      const rows = sqlite.prepare(
        `UPDATE embedding_jobs
         SET status = 'processing', updated_at = ?
         WHERE id IN (
           SELECT id FROM embedding_jobs
           WHERE status = 'pending' AND attempts < max_attempts
           ORDER BY created_at ASC LIMIT ?
         )
         RETURNING id, content_hash, status, attempts, max_attempts, error, created_at, updated_at`
      ).all(now, batchSize) as Array<{
        id: number;
        content_hash: string;
        status: string;
        attempts: number;
        max_attempts: number;
        error: string | null;
        created_at: string;
        updated_at: string;
      }>;
      return rows.map(r => ({
        id: r.id,
        contentHash: r.content_hash,
        status: r.status,
        attempts: r.attempts,
        maxAttempts: r.max_attempts,
        error: r.error,
        createdAt: r.created_at,
        updatedAt: r.updated_at,
      }));
    }

    export function markJobComplete(sqlite: Database.Database, jobId: number): void {
      const now = new Date().toISOString();
      sqlite.prepare(
        `UPDATE embedding_jobs SET status = 'complete', updated_at = ? WHERE id = ?`
      ).run(now, jobId);
    }

    export function markJobFailed(sqlite: Database.Database, jobId: number, error: string): void {
      const now = new Date().toISOString();
      sqlite.prepare(
        `UPDATE embedding_jobs SET status = 'failed', error = ?, attempts = attempts + 1, updated_at = ? WHERE id = ?`
      ).run(error, now, jobId);
    }

    /**
     * Reset failed jobs back to pending for retry (only if attempts < max_attempts).
     */
    export function resetRetryableJobs(sqlite: Database.Database): number {
      const now = new Date().toISOString();
      const result = sqlite.prepare(
        `UPDATE embedding_jobs SET status = 'pending', updated_at = ?
         WHERE status = 'failed' AND attempts < max_attempts`
      ).run(now);
      return result.changes;
    }
    ```

    **2. Create `packages/api/src/services/embedding-service.ts`:**
    ```typescript
    import { createHash } from 'node:crypto';
    import type Database from 'better-sqlite3';
    import { upsertSearchContent, hasEmbedding, insertEmbedding } from '../db/queries/vector-search.js';
    import { enqueueEmbeddingJob, claimPendingJobs, markJobComplete, markJobFailed } from '../db/queries/embedding-jobs.js';
    import { generateEmbedding as lmGenerateEmbedding, createLmStudioProvider } from './lm-studio.js';
    import { getLmStudioStatus } from './lm-studio.js';

    /**
     * Compute content hash using SHA-256 with CRLF normalization.
     * Reuses existing Phase 24 pattern.
     */
    export function contentHash(text: string): string {
      const normalized = text.replace(/\r\n/g, '\n');
      return createHash('sha256').update(normalized, 'utf-8').digest('hex');
    }

    /**
     * Index content for both FTS5 (existing) and vector embedding (new).
     * Content-addressable: if hash already exists, skips (D-03).
     * Returns the content hash.
     */
    export function indexContentForEmbedding(
      sqlite: Database.Database,
      data: {
        content: string;
        sourceType: string;
        sourceId: string;
        projectSlug: string | null;
        createdAt: string;
      }
    ): string {
      const hash = contentHash(data.content);
      const isNew = upsertSearchContent(sqlite, {
        contentHash: hash,
        content: data.content,
        sourceType: data.sourceType,
        sourceId: data.sourceId,
        projectSlug: data.projectSlug,
        createdAt: data.createdAt,
      });
      if (isNew) {
        enqueueEmbeddingJob(sqlite, hash);
      }
      return hash;
    }

    /**
     * Process pending embedding jobs in a batch.
     * Claims jobs, generates embeddings via LM Studio, inserts into vec0.
     * Gracefully skips if LM Studio unavailable.
     */
    export async function processEmbeddingJobs(
      sqlite: Database.Database,
      lmStudioUrl: string,
      batchSize: number = 10
    ): Promise<{ processed: number; failed: number }> {
      const status = getLmStudioStatus();
      if (status.health !== 'ready') {
        return { processed: 0, failed: 0 };
      }

      const jobs = claimPendingJobs(sqlite, batchSize);
      if (jobs.length === 0) {
        return { processed: 0, failed: 0 };
      }

      const provider = createLmStudioProvider(lmStudioUrl);
      let processed = 0;
      let failed = 0;

      for (const job of jobs) {
        try {
          if (hasEmbedding(sqlite, job.contentHash)) {
            markJobComplete(sqlite, job.id);
            processed++;
            continue;
          }

          const content = sqlite.prepare(
            'SELECT content FROM search_content WHERE content_hash = ?'
          ).get(job.contentHash) as { content: string } | undefined;

          if (!content) {
            markJobFailed(sqlite, job.id, 'Content not found in search_content');
            failed++;
            continue;
          }

          const embedding = await lmGenerateEmbedding(provider, content.content);
          insertEmbedding(sqlite, job.contentHash, embedding);
          markJobComplete(sqlite, job.id);
          processed++;
        } catch (err) {
          const message = err instanceof Error ? err.message : 'Unknown error';
          markJobFailed(sqlite, job.id, message);
          failed++;
        }
      }

      return { processed, failed };
    }
    ```
