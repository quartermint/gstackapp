    ```typescript
    import { generateText, Output } from 'ai';
    import { z } from 'zod';
    import { createLmStudioProvider, getLmStudioStatus } from './lm-studio.js';

    const RERANK_TIMEOUT_MS = 2000;
    const RERANK_TOP_N = 20;
    const TOP_TIER_COUNT = 5;
    const TOP_TIER_RRF_WEIGHT = 0.75;
    const TOP_TIER_RERANK_WEIGHT = 0.25;
    const DEEP_TIER_RRF_WEIGHT = 0.40;
    const DEEP_TIER_RERANK_WEIGHT = 0.60;

    export interface RerankCandidate {
      contentHash: string;
      content: string;
      rrfScore: number;
    }

    export interface RerankedResult {
      contentHash: string;
      finalScore: number;
    }

    const rerankScoreSchema = z.object({
      relevance: z.number().min(0).max(1).describe("Relevance score 0-1 for how well the document matches the query"),
    });

    /**
     * Cross-encoder reranking via LM Studio (D-08).
     * Position-aware blending: top results trust RRF more (75/25),
     * deep results trust reranker more (40/60).
     * Returns original order on timeout or LM Studio unavailable.
     */
    export async function rerankResults(
      query: string,
      candidates: RerankCandidate[],
      lmStudioUrl: string
    ): Promise<RerankedResult[]> {
      // Skip if no candidates or too few to rerank
      if (candidates.length <= 1) {
        return candidates.map(c => ({ contentHash: c.contentHash, finalScore: c.rrfScore }));
      }

      // Skip if LM Studio unavailable
      const status = getLmStudioStatus();
      if (status.health !== 'ready') {
        return candidates.map(c => ({ contentHash: c.contentHash, finalScore: c.rrfScore }));
      }

      // Only rerank top N candidates
      const toRerank = candidates.slice(0, RERANK_TOP_N);
      const remainder = candidates.slice(RERANK_TOP_N);

      try {
        const provider = createLmStudioProvider(lmStudioUrl);
        const controller = new AbortController();
        const timeoutId = setTimeout(() => controller.abort(), RERANK_TIMEOUT_MS);

        const rerankScores: number[] = [];

        for (const candidate of toRerank) {
          try {
            const { output } = await generateText({
              model: provider.chatModel('qwen3-coder'),
              output: Output.object({ schema: rerankScoreSchema }),
              prompt: `Rate the relevance of this document to the search query on a scale of 0 to 1.
    Query: "${query}"
    Document: "${candidate.content.slice(0, 500)}"
    Return a relevance score between 0.0 (not relevant) and 1.0 (highly relevant).`,
              abortSignal: controller.signal,
            });
            rerankScores.push(output?.relevance ?? 0.5);
          } catch {
            rerankScores.push(0.5); // Neutral score on individual failure
          }
        }

        clearTimeout(timeoutId);

        // Position-aware blending
        const reranked = toRerank.map((candidate, i) => {
          const rerankScore = rerankScores[i] ?? 0.5;
          const isTopTier = i < TOP_TIER_COUNT;
          const rrfWeight = isTopTier ? TOP_TIER_RRF_WEIGHT : DEEP_TIER_RRF_WEIGHT;
          const rerankWeight = isTopTier ? TOP_TIER_RERANK_WEIGHT : DEEP_TIER_RERANK_WEIGHT;

          return {
            contentHash: candidate.contentHash,
            finalScore: (rrfWeight * candidate.rrfScore) + (rerankWeight * rerankScore),
          };
        });

        // Sort reranked by final score descending
        reranked.sort((a, b) => b.finalScore - a.finalScore);

        // Append remainder unchanged
        const remainderResults = remainder.map(c => ({
          contentHash: c.contentHash,
          finalScore: c.rrfScore,
        }));

        return [...reranked, ...remainderResults];
      } catch {
        // Timeout or other error: return original order
        return candidates.map(c => ({ contentHash: c.contentHash, finalScore: c.rrfScore }));
      }
    }
    ```

    ```typescript
    import { vectorKnnSearch, enrichVectorResults } from './vector-search.js';
    import { fuseResults, rankByFusion, type FusionCandidate } from '../services/rrf-fusion.js';
    import { contentHash } from '../services/embedding-service.js';
    import { generateEmbedding, createLmStudioProvider, getLmStudioStatus } from '../services/lm-studio.js';
    import type { ProcessedQuery } from '../services/ai-query-rewriter.js';

    export interface HybridSearchResult {
      content: string;
      snippet: string;
      sourceType: "capture" | "commit" | "project" | "knowledge";
      sourceId: string;
      projectSlug: string | null;
      score: number;          // fused RRF score
      createdAt: string;
      projectContext?: string; // D-06: context annotation
    }

    export interface HybridSearchOptions {
      limit?: number;
      sourceType?: "capture" | "commit" | "project" | "knowledge";
      projectSlug?: string;
      dateAfter?: string;
      dateBefore?: string;
      lmStudioUrl?: string;
    }

    /**
     * Hybrid search pipeline:
     * 1. BM25 via FTS5 (always runs)
     * 2. Vector via sqlite-vec KNN (runs if embeddings exist + query embedding available)
     * 3. RRF fusion (combines BM25 + vector rankings)
     * 4. Context annotations (D-06)
     *
     * Degradation levels:
     *   Level 0: BM25 + Vector + RRF + expanded variants
     *   Level 1: BM25 + Vector + RRF (no expansion)
     *   Level 2: BM25 only (no vector)
     */
    export async function hybridSearch(
      sqlite: Database.Database,
      processed: ProcessedQuery,
      options: HybridSearchOptions = {}
    ): Promise<HybridSearchResult[]> {
      const { limit = 20, lmStudioUrl } = options;
      const fetchLimit = limit * 3; // Over-fetch for fusion

      // --- Step 1: BM25 results (always available) ---
      const bm25Original = searchUnified(sqlite, processed.ftsQuery, {
        ...options, limit: fetchLimit
      });

      // BM25 from expanded lex variants (if available)
      const bm25Expanded: UnifiedSearchResult[] = [];
      if (processed.lexVariants) {
        for (const variant of processed.lexVariants) {
          const sanitized = sanitizeFtsQuery(variant);
          const results = searchUnified(sqlite, sanitized, { ...options, limit: fetchLimit });
          bm25Expanded.push(...results);
        }
      }

      // --- Step 2: Vector results (if LM Studio available) ---
      let vectorOriginal: { contentHash: string; distance: number }[] = [];
      let vectorExpanded: { contentHash: string; distance: number }[] = [];

      const status = getLmStudioStatus();
      if (status.health === 'ready' && lmStudioUrl) {
        try {
          const provider = createLmStudioProvider(lmStudioUrl);

          // Query embedding with 2-second timeout
          const queryEmb = await Promise.race([
            generateEmbedding(provider, processed.ftsQuery.replace(/"/g, '')),
            new Promise<null>((_, reject) => setTimeout(() => reject(new Error('timeout')), 2000)),
          ]);

          if (queryEmb) {
            vectorOriginal = vectorKnnSearch(sqlite, queryEmb as Float32Array, fetchLimit);
          }

          // Expanded vec variants
          if (processed.vecVariants) {
            for (const variant of processed.vecVariants) {
              try {
                const variantEmb = await Promise.race([
                  generateEmbedding(provider, variant),
                  new Promise<null>((_, reject) => setTimeout(() => reject(new Error('timeout')), 2000)),
                ]);
                if (variantEmb) {
                  const results = vectorKnnSearch(sqlite, variantEmb as Float32Array, fetchLimit);
                  vectorExpanded.push(...results);
                }
              } catch { /* skip this variant */ }
            }
          }
        } catch { /* vector search unavailable, continue with BM25 only */ }
      }

      // --- Step 3: Build content hash map for BM25 results ---
      // BM25 results don't have content hashes -- compute them
      const bm25HashMap = new Map<string, UnifiedSearchResult>();
      for (const r of [...bm25Original, ...bm25Expanded]) {
        const hash = contentHash(r.content);
        if (!bm25HashMap.has(hash)) {
          bm25HashMap.set(hash, r);
        }
      }

      // --- Step 4: RRF Fusion (D-04) ---
      const rankedLists: FusionCandidate[][] = [];

      // BM25 original (weight 2.0 per D-04)
      rankedLists.push(
        bm25Original.map((r, i) => ({
          contentHash: contentHash(r.content),
          rank: i,
          weight: 2.0,
        }))
      );

      // Vector original (weight 2.0 per D-04)
      if (vectorOriginal.length > 0) {
        rankedLists.push(
          vectorOriginal.map((r, i) => ({
            contentHash: r.contentHash,
            rank: i,
            weight: 2.0,
          }))
        );
      }

      // BM25 expanded (weight 1.0)
      if (bm25Expanded.length > 0) {
        // Deduplicate expanded results
        const seen = new Set<string>();
        const dedupedExpanded = bm25Expanded.filter(r => {
          const hash = contentHash(r.content);
          if (seen.has(hash)) return false;
          seen.add(hash);
          return true;
        });
        rankedLists.push(
          dedupedExpanded.map((r, i) => ({
            contentHash: contentHash(r.content),
            rank: i,
            weight: 1.0,
          }))
        );
      }

      // Vector expanded (weight 1.0)
      if (vectorExpanded.length > 0) {
        const seen = new Set<string>();
        const dedupedVec = vectorExpanded.filter(r => {
          if (seen.has(r.contentHash)) return false;
          seen.add(r.contentHash);
          return true;
        });
        rankedLists.push(
          dedupedVec.map((r, i) => ({
            contentHash: r.contentHash,
            rank: i,
            weight: 1.0,
          }))
        );
      }

      const fusedScores = fuseResults(rankedLists);
      const rankedHashes = rankByFusion(fusedScores);

      // --- Step 5: Enrich results with metadata ---
      // Get metadata from both BM25 hash map and search_content table
      const vectorMeta = enrichVectorResults(sqlite, rankedHashes.filter(h => !bm25HashMap.has(h)));

      // --- Step 6: Build final results with context annotations (D-06) ---
      const projectContextCache = new Map<string, string>();
      const results: HybridSearchResult[] = [];

      for (const hash of rankedHashes.slice(0, limit)) {
        const score = fusedScores.get(hash) ?? 0;
        const bm25Meta = bm25HashMap.get(hash);
        const vecMeta = vectorMeta.get(hash);
        const meta = bm25Meta ?? vecMeta;

        if (!meta) continue;

        // Context annotation (D-06)
        let projectContext: string | undefined;
        const slug = 'projectSlug' in meta ? meta.projectSlug : null;
        if (slug && !projectContextCache.has(slug)) {
          const ctx = sqlite.prepare(
            `SELECT p.tagline, pk.content FROM projects p
             LEFT JOIN project_knowledge pk ON pk.project_slug = p.slug
             WHERE p.slug = ?`
          ).get(slug) as { tagline: string | null; content: string | null } | undefined;
          if (ctx) {
            const excerpt = ctx.content ? ctx.content.slice(0, 200).split('\n').slice(0, 3).join(' ') : '';
            projectContextCache.set(slug, ctx.tagline ? `${ctx.tagline}${excerpt ? ' — ' + excerpt : ''}` : excerpt);
          }
        }
        if (slug) {
          projectContext = projectContextCache.get(slug);
        }

        const sourceType = ('sourceType' in meta ? meta.sourceType : meta.sourceType) as HybridSearchResult['sourceType'];
        const content = 'content' in meta ? meta.content : '';
        const snippet = 'snippet' in meta && meta.snippet ? meta.snippet : content.slice(0, 120);

        results.push({
          content,
          snippet,
          sourceType,
          sourceId: 'sourceId' in meta ? meta.sourceId : '',
          projectSlug: slug,
          score,
          createdAt: 'createdAt' in meta ? String(meta.createdAt) : '',
          projectContext,
        });
      }

      return results;
    }
    ```

    ```typescript
    import { Hono } from "hono";
    import { zValidator } from "@hono/zod-validator";
    import { searchQuerySchema } from "@mission-control/shared";
    import { hybridSearch } from "../db/queries/search.js";
    import { processSearchQuery } from "../services/ai-query-rewriter.js";
    import { rerankResults } from "../services/reranker.js";
    import { listProjects } from "../db/queries/projects.js";
    import type { DatabaseInstance } from "../db/index.js";
    import type { MCConfig } from "../lib/config.js";

    export function createSearchRoutes(
      getInstance: () => DatabaseInstance,
      getConfig?: () => MCConfig
    ) {
      return new Hono().get(
        "/search",
        zValidator("query", searchQuerySchema),
        async (c) => {
          const { q, limit } = c.req.valid("query");
          const { db, sqlite } = getInstance();
          const config = getConfig?.();
          const lmStudioUrl = config?.lmStudio?.url ?? "http://100.123.8.125:1234";

          // Get projects list for query expansion context
          const projects = listProjects(db).map(p => ({
            slug: p.slug,
            name: p.name,
            tagline: p.tagline,
          }));

          // Process query: expansion + filters
          const processed = await processSearchQuery(q, projects);

          // Hybrid search: BM25 + vector + RRF
          const hybridResults = await hybridSearch(sqlite, processed, {
            limit: limit ?? 20,
            sourceType: processed.filters?.type ?? undefined,
            projectSlug: processed.filters?.project ?? undefined,
            dateAfter: processed.filters?.dateAfter ?? undefined,
            dateBefore: processed.filters?.dateBefore ?? undefined,
            lmStudioUrl,
          });

          // Optional reranking (D-08)
          const candidates = hybridResults.map(r => ({
            contentHash: r.sourceId,  // Use sourceId as key
            content: r.content,
            rrfScore: r.score,
          }));
          const reranked = await rerankResults(q, candidates, lmStudioUrl);

          // Map back to results preserving order
          const rerankedMap = new Map(reranked.map((r, i) => [r.contentHash, { index: i, finalScore: r.finalScore }]));
          const sortedResults = [...hybridResults].sort((a, b) => {
            const aInfo = rerankedMap.get(a.sourceId);
            const bInfo = rerankedMap.get(b.sourceId);
            return (aInfo?.index ?? 0) - (bInfo?.index ?? 0);
          });

          // Build response
          const resultsWithId = sortedResults.map(r => ({
            id: r.sourceId,
            content: r.content,
            snippet: r.snippet,
            sourceType: r.sourceType,
            sourceId: r.sourceId,
            projectSlug: r.projectSlug,
            score: rerankedMap.get(r.sourceId)?.finalScore ?? r.score,
            rank: rerankedMap.get(r.sourceId)?.finalScore ?? r.score,  // backward compat
            createdAt: r.createdAt,
            projectContext: r.projectContext,
          }));

          return c.json({
            results: resultsWithId,
            query: q,
            rewrittenQuery: processed.rewritten ? processed.ftsQuery : null,
            filters: processed.filters,
          });
        }
      );
    }
    ```
