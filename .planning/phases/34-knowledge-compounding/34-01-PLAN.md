```typescript
// -- Knowledge Compounding (Phase 34) --

export const solutions = sqliteTable(
  "solutions",
  {
    id: text("id").primaryKey(),
    sessionId: text("session_id"),
    projectSlug: text("project_slug"),
    title: text("title").notNull(),
    content: text("content").notNull(),
    contentHash: text("content_hash").notNull(),
    module: text("module"),
    problemType: text("problem_type", {
      enum: ["bug_fix", "architecture", "performance", "integration", "configuration", "testing", "deployment"],
    }),
    symptoms: text("symptoms"),
    rootCause: text("root_cause"),
    tagsJson: text("tags_json"),
    severity: text("severity", {
      enum: ["low", "medium", "high", "critical"],
    }).default("medium"),
    status: text("status", {
      enum: ["candidate", "accepted", "dismissed"],
    }).notNull().default("candidate"),
    referenceCount: integer("reference_count").notNull().default(0),
    createdAt: integer("created_at", { mode: "timestamp" }).notNull(),
    updatedAt: integer("updated_at", { mode: "timestamp" }).notNull(),
    reviewedAt: integer("reviewed_at", { mode: "timestamp" }),
  },
  (table) => [
    index("solutions_project_slug_idx").on(table.projectSlug),
    index("solutions_status_idx").on(table.status),
    index("solutions_problem_type_idx").on(table.problemType),
    index("solutions_session_id_idx").on(table.sessionId),
    uniqueIndex("solutions_content_hash_uniq").on(table.contentHash),
  ]
);

export const solutionReferences = sqliteTable(
  "solution_references",
  {
    id: text("id").primaryKey(),
    solutionId: text("solution_id").notNull(),
    sessionId: text("session_id").notNull(),
    referenceType: text("reference_type", {
      enum: ["startup_banner", "search_result", "mcp_query"],
    }).notNull(),
    createdAt: integer("created_at", { mode: "timestamp" }).notNull(),
  },
  (table) => [
    index("sol_ref_solution_id_idx").on(table.solutionId),
    index("sol_ref_session_id_idx").on(table.sessionId),
  ]
);
```

1. **queries/solutions.ts** -- Create CRUD functions following knowledge.ts patterns:

   - `createSolution(db, data)` -- Insert with crypto.randomUUID() for id, return the created row. Data includes: sessionId, projectSlug, title, content, contentHash, module, problemType, symptoms, rootCause, tagsJson, severity, status (default "candidate").
   - `getSolution(db, id)` -- Get by id, throw notFound if missing.
   - `listSolutions(db, query)` -- Filter by projectSlug, status, problemType. Support limit/offset. Return { solutions, total }.
   - `updateSolutionStatus(db, id, status)` -- Update status + reviewedAt timestamp. Return updated solution.
   - `updateSolutionMetadata(db, id, metadata)` -- Update enrichment fields (module, problemType, symptoms, rootCause, tagsJson, severity, title). Return updated solution.
   - `solutionExistsForHash(db, contentHash)` -- Boolean check for dedup.
   - `getRelevantSolutions(db, projectSlug, limit)` -- Get accepted solutions for a project, ordered by referenceCount DESC then createdAt DESC. This is for the MCP startup banner (COMP-03).
   - `recordSolutionReference(db, solutionId, sessionId, referenceType)` -- Insert into solution_references, increment referenceCount on solutions table. Use crypto.randomUUID() for reference id.
   - `getCompoundScore(db)` -- Calculate: totalSolutions (all accepted), referencedSolutions (accepted with referenceCount > 0), totalReferences (sum of all reference_count), reuseRate (referenced/total). For weeklyTrend: query solution_references grouped by ISO week, last 8 weeks. Return CompoundScore shape.

   - `GET /solutions/compound-score` -- getCompoundScore().
   - `GET /solutions/relevant` -- getRelevantSolutions(). Requires `cwd` query param, resolves to project slug via resolveProjectFromCwd. Returns top 3 accepted solutions. Import resolveProjectFromCwd from session-service and getConfig parameter.
