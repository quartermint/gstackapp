**CaptureClassifier.swift (new file: MissionControl/Services/):**

The entire file is wrapped in `#if canImport(FoundationModels)`. Implements CaptureClassifierProtocol using Foundation Models. Per D-03, manages the 4096 token context window.

```swift
#if canImport(FoundationModels)
import Foundation
import FoundationModels

@available(iOS 26, *)
@MainActor
@Observable
final class FoundationModelsCaptureClassifier: CaptureClassifierProtocol, @unchecked Sendable {
    private let model = SystemLanguageModel.default
    private let maxTokens = 4096
    private let safeTokenLimit = 3500  // 70% threshold per D-03

    var isAvailable: Bool {
        switch model.availability {
        case .available: return true
        default: return false
        }
    }

    /// Prewarm the model for faster first classification.
    func prewarm() {
        guard isAvailable else { return }
        let session = LanguageModelSession()
        session.prewarm()
    }

    func classify(
        content: String,
        projects: [(slug: String, name: String)]
    ) async throws -> DeviceClassificationResult? {
        guard isAvailable else { return nil }
        guard !projects.isEmpty else { return nil }
        guard !content.trimmingCharacters(in: .whitespacesAndNewlines).isEmpty else { return nil }

        let projectListStr = projects.map { "\($0.slug): \($0.name)" }.joined(separator: "\n")

        // EDGE-06: Context window management (per D-03)
        let captureContent: String
        let contentTokenEstimate = estimateTokens(content)
        let overheadEstimate = 200 + estimateTokens(projectListStr) + 500  // system + projects + response

        if contentTokenEstimate + overheadEstimate > safeTokenLimit {
            // Summarize long content before classifying
            let summarizer = LanguageModelSession()
            let summary = try await summarizer.respond(
                to: "Summarize this text concisely in 2-3 sentences, preserving key topics and any project references:\n\n\(content)"
            )
            captureContent = summary.content
        } else {
            captureContent = content
        }

        let session = LanguageModelSession {
            """
            You are a capture classifier for Mission Control. Given capture text and a project list, \
            classify the capture to the most relevant project. If the capture doesn't clearly match \
            any project, set projectSlug to "unassigned" and confidence to a low value.

            Available projects:
            \(projectListStr)
            """
        }

        let response = try await session.respond(
            to: "Classify this capture:\n\n\(captureContent)",
            generating: CaptureClassification.self
        )

        let classification = response.content
        return DeviceClassificationResult(
            projectSlug: classification.projectSlug == "unassigned" ? nil : classification.projectSlug,
            confidence: classification.confidence,
            extractionType: classification.extractionType.rawValue,
            reasoning: classification.reasoning,
            classifiedAt: Date(),
            classifiedOnDevice: true
        )
    }

    /// Estimate token count using word-count heuristic (per research: word_count / 0.75).
    /// Apple does not expose a public tokenizer.
    func estimateTokens(_ text: String) -> Int {
        let wordCount = text.split(separator: " ").count
        return max(1, Int(Double(wordCount) / 0.75))
    }
}
#endif
```

Key implementation decisions:
- `@MainActor` because Foundation Models requires main actor context
- `@unchecked Sendable` because @Observable + @MainActor combo needs explicit Sendable for protocol conformance
- New LanguageModelSession per classification (one-shot; sessions carry context we don't need)
- `estimateTokens` is public (not private) so tests can verify the heuristic
- Catches exceededContextWindowSize error implicitly -- it propagates as throws to caller, who falls back to server

```swift
/// EDGE-02: Classify unclassified captures on-device before syncing.
/// Skips captures already classified or where classifier is unavailable.
func classifyPendingCaptures(connectionMonitor: ConnectionMonitor) async {
    guard classifier.isAvailable else { return }
    guard let container = modelContainer else { return }

    let context = ModelContext(container)
    let descriptor = FetchDescriptor<QueuedCapture>(
        predicate: #Predicate<QueuedCapture> { capture in
            (capture.syncStatus == "pending" || capture.syncStatus == "failed")
            && capture.deviceClassifiedAt == nil
        }
    )

    guard let unclassified = try? context.fetch(descriptor) else { return }
    guard !unclassified.isEmpty else { return }

    // Read project list from cache (same source as share extension picker)
    let projects = cachedProjectList()
    guard !projects.isEmpty else { return }

    for capture in unclassified {
        let content = capture.type == "voice" ? (capture.transcriptionText ?? capture.rawContent) : capture.rawContent

        // EDGE-05: Stamp device context metadata
        capture.timeOfDay = DeviceContext.currentTimeOfDay()
        capture.connectivityState = DeviceContext.connectivityState(
            isNetworkAvailable: connectionMonitor.isNetworkAvailable,
            isMCReachable: connectionMonitor.isMCReachable
        )

        do {
            if let result = try await classifier.classify(content: content, projects: projects) {
                capture.deviceProjectSlug = result.projectSlug
                capture.deviceConfidence = result.confidence
                capture.deviceExtractionType = result.extractionType
                capture.deviceReasoning = result.reasoning
                capture.deviceClassifiedAt = result.classifiedAt
            }
        } catch {
            // Classification failure is non-fatal -- capture syncs without device hint
            // Server enrichment handles classification as fallback
        }
    }

    try? context.save()
}

/// Read cached project list from App Group UserDefaults.
private func cachedProjectList() -> [(slug: String, name: String)] {
    guard let defaults = UserDefaults(suiteName: AppConstants.appGroupID),
          let data = defaults.data(forKey: AppConstants.projectListUserDefaultsKey),
          let list = try? JSONDecoder().decode([[String: String]].self, from: data) else {
        return []
    }
    return list.compactMap { dict in
        guard let slug = dict["slug"], let name = dict["name"] else { return nil }
        return (slug: slug, name: name)
    }
}
```
