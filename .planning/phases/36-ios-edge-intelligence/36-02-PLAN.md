**1. CaptureClassifierProtocol.swift (new file: MissionControl/Services/):**

Per D-01, Foundation Models is an enhancement layer. Protocol enables graceful degradation.

```swift
import Foundation

/// Protocol abstracting capture classification for testability and availability-based swapping.
/// Devices without Foundation Models use NoOpCaptureClassifier; iOS 26+ with Apple Intelligence
/// uses FoundationModelsCaptureClassifier (created in Plan 03).
protocol CaptureClassifierProtocol: Sendable {
    var isAvailable: Bool { get }
    func classify(content: String, projects: [(slug: String, name: String)]) async throws -> DeviceClassificationResult?
}

/// Fallback classifier for devices without Foundation Models (iOS 17-25, or iOS 26 without Apple Intelligence).
/// Returns nil -- server handles classification. Per D-01: captures work without on-device enrichment.
final class NoOpCaptureClassifier: CaptureClassifierProtocol, @unchecked Sendable {
    let isAvailable = false
    func classify(content: String, projects: [(slug: String, name: String)]) async throws -> DeviceClassificationResult? {
        return nil
    }
}
```

/// Constrained decoding struct for Foundation Models (per D-02).
/// The model is forced to produce this exact structure at the token level.
@available(iOS 26, *)
@Generable
struct CaptureClassification {
    @Guide(description: "The project slug this capture belongs to, or 'unassigned' if unclear")
    var projectSlug: String

    @Guide(description: "Confidence score from 0.0 to 1.0", .range(0.0...1.0))
    var confidence: Double

    @Guide(description: "The type of capture content")
    var extractionType: CaptureExtractionType

    @Guide(description: "Brief reasoning for the classification in one sentence")
    var reasoning: String
}

/// Extraction types matching server-side extractionTypeEnum.
@available(iOS 26, *)
@Generable
enum CaptureExtractionType: String, Codable {
    case projectRef = "project_ref"
    case actionItem = "action_item"
    case idea
    case link
    case question

**3. DeviceContext.swift (new file: MissionControl/Models/):**

Per D-07 and EDGE-05. Compute device context metadata.

```swift
import Foundation
import Network

/// Computes device context metadata for captures (EDGE-05, D-07).
enum DeviceContext {
    /// Classify current time into period.
    static func currentTimeOfDay() -> String {
        let hour = Calendar.current.component(.hour, from: Date())
        switch hour {
        case 5..<12: return "morning"
        case 12..<17: return "afternoon"
        case 17..<21: return "evening"
        default: return "night"
        }
    }

    /// Map NWPath.Status to connectivity string.
    static func connectivityState(isNetworkAvailable: Bool, isMCReachable: Bool) -> String {
        if !isNetworkAvailable { return "offline" }
        if isMCReachable { return "tailscale" }
        return "online"  // network available but MC not reachable (no Tailscale)
    }
}
```

import XCTest
@testable import MissionControl

final class DeviceContextTests: XCTestCase {

    func testTimeOfDayMorning() {
        // Test the range boundaries
        // Note: DeviceContext.currentTimeOfDay() uses current time,
        // so we test the classification logic by verifying it returns valid values
        let result = DeviceContext.currentTimeOfDay()
        XCTAssertTrue(["morning", "afternoon", "evening", "night"].contains(result))
    }

    func testConnectivityStateOffline() {
        let result = DeviceContext.connectivityState(isNetworkAvailable: false, isMCReachable: false)
        XCTAssertEqual(result, "offline")
    }

    func testConnectivityStateTailscale() {
        let result = DeviceContext.connectivityState(isNetworkAvailable: true, isMCReachable: true)
        XCTAssertEqual(result, "tailscale")
    }

    func testConnectivityStateOnlineNoTailscale() {
        let result = DeviceContext.connectivityState(isNetworkAvailable: true, isMCReachable: false)
        XCTAssertEqual(result, "online")
    }

    func testConnectivityStateOfflineOverridesMCReachable() {
        // Edge case: network down but MC marked reachable (stale state)
        let result = DeviceContext.connectivityState(isNetworkAvailable: false, isMCReachable: true)
        XCTAssertEqual(result, "offline")
    }
}
