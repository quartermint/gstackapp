1. **Create session hooks** (`packages/web/src/hooks/useSession.ts`):

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'

interface Session {
  id: string; title: string | null; status: string;
  messageCount: number; lastMessageAt: string | null;
  projectPath: string | null; createdAt: string;
}

interface SessionMessage {
  id: string; role: 'user' | 'assistant' | 'system';
  content: string; hasToolCalls: boolean; createdAt: string;
}

export function useSessions() {
  return useQuery({
    queryKey: ['sessions'],
    queryFn: () => fetch('/api/sessions').then(r => r.json()).then(d => d.sessions as Session[]),
  })
}

export function useSessionDetail(sessionId: string | null) {
  return useQuery({
    queryKey: ['session', sessionId],
    queryFn: () => fetch(`/api/sessions/${sessionId}`).then(r => r.json()),
    enabled: !!sessionId,
  })
}

export function useCreateSession() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (opts: { title?: string; projectPath?: string }) =>
      fetch('/api/sessions', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(opts) }).then(r => r.json()),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['sessions'] }),
  })
}
```

2. **Create agent stream hook** (`packages/web/src/hooks/useAgentStream.ts`):

Manage the SSE connection to /api/agent/stream. State:
- `messages: Array<{ role, content, toolCalls? }>` — accumulated message list
- `streamingText: string` — currently streaming text (cleared on turn_complete)
- `activeTools: Map<string, { name, input?, output?, isError? }>` — tools currently executing
- `isStreaming: boolean` — whether stream is active
- `compacted: boolean` — whether compaction has fired

`sendMessage(prompt: string, sessionId?: string)` function:
- Opens EventSource to `/api/agent/stream?prompt=${encodeURIComponent(prompt)}&sessionId=${sessionId}`
- Listens for event types: text_delta, tool_start, tool_result, turn_complete, result, compact, error
- Accumulates text_delta into streamingText
- On tool_start: add to activeTools
- On tool_result: update tool in activeTools with output
- On turn_complete: move streamingText into messages array as assistant message, clear streamingText
- On result: set isStreaming false, close EventSource, return sessionId for metadata
- On error: set isStreaming false, add error to messages
- On compact: set compacted true
